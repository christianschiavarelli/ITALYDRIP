<script>
// Carrello persistente con localStorage
let cart = []

// Carica il carrello da localStorage all'avvio
window.onload = function(){
  const savedCart = localStorage.getItem('cart')
  if(savedCart){
    cart = JSON.parse(savedCart)
  }
  renderCart()
}

// Salva il carrello su localStorage
function saveCart(){
  localStorage.setItem('cart', JSON.stringify(cart))
}

// Aggiungi prodotto al carrello
function addToCart(name, price, sizeId, qtyId){
  let size = document.getElementById(sizeId).value
  let qty = Math.max(1, parseInt(document.getElementById(qtyId).value) || 1)
  cart.push({name, price, qty, size})
  saveCart()
  renderCart()
}

// Render del carrello
function renderCart(){
  let div = document.getElementById("cartItems")
  div.innerHTML = ""
  let total = 0

  if(cart.length === 0){
    div.innerHTML = "<p>Il carrello è vuoto.</p>"
  } else {
    cart.forEach((item, index) => {
      total += item.price * item.qty
      div.innerHTML += `<div class="cart-item">
        <span>${item.name}</span>
        <span>${item.size}</span>
        <input type="number" value="${item.qty}" min="1" onchange="updateQty(${index}, this.value)">
        <span>€${item.price * item.qty}</span>
        <span class="remove-btn" onclick="removeItem(${index})">X</span>
      </div>`
    })
  }

  document.getElementById("total").innerText = total
}

// Aggiorna quantità di un prodotto
function updateQty(index, newQty){
  newQty = Math.max(1, parseInt(newQty) || 1)
  cart[index].qty = newQty
  saveCart()
  renderCart()
}

// Rimuovi prodotto dal carrello
function removeItem(index){
  cart.splice(index, 1)
  saveCart()
  renderCart()
}
</script>
